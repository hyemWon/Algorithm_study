# vector 이용해서 풀기
```cpp
#include <iostream>
#include <algorithm>
#include <vector>
#include <queue>

using namespace std;

int main()
{
	int N, K;
	cin >> N >> K;
	vector<int>v;
	vector<int>answer;
	for (int i = 1; i <= N; i++) { v.push_back(i); }
	int cnt = 1;
	while (!v.empty()) {
		if (cnt == K) {
			answer.push_back(v[0]);
			v.erase(v.begin());
			cnt = 1;
		}
		else {
			v.push_back(v[0]);
			v.erase(v.begin());
			cnt++;
		}
	}
	cout << "<";
	for (int i = 0; i < answer.size()-1; i++) {
		cout << answer[i] << "," <<" ";
	}
	cout << answer[answer.size() - 1] << ">" << endl;
}
```
